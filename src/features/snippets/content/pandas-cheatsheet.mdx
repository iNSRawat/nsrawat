```python
import pandas as pd

# Read data
df = pd.read_csv('data.csv')

# Basic info
df.head()
df.info()
df.describe()

# Filter rows
df[df['column'] > value]
df.query('column > @value')

# Group by
df.groupby('category').agg({'value': ['mean', 'sum']})
```
